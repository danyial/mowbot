    import * as ROSLIB from "roslib";
    import { getRos } from "./ros-client";

    /**
     * Phase 4 Plan 04-02 — Reset the slam_toolbox map in place (no container restart).
     * Upstream service spec: slam_toolbox/srv/Reset — empty request, empty response.
     * https://github.com/SteveMacenski/slam_toolbox/blob/humble/srv/Reset.srv
     */
    export function callSlamReset(): Promise<void> {
      return new Promise((resolve, reject) => {
        const ros = getRos();
        if (!ros) {
          reject(new Error("ROS not connected"));
          return;
        }
        const svc = new ROSLIB.Service({
          ros,
          name: "/slam_toolbox/reset",
          serviceType: "slam_toolbox/srv/Reset",
        });
        const req = new ROSLIB.ServiceRequest({});
        svc.callService(
          req,
          () => resolve(),
          (err: unknown) => reject(err instanceof Error ? err : new Error(String(err)))
        );
      });
    }

    **web/components/lidar/map-bitmap.tsx** — NEW. Canvas 2D component, standalone-only (no anchored/Leaflet mode — `/map` does not render MapBitmap). Core pattern from RESEARCH.md lines ~222-258:

    Key design:
    - One offscreen "backing" canvas sized `info.width × info.height` pixels (1 grid cell = 1 px). `putImageData` once per new `useMapStore.latest` identity. This is the expensive step, done only on map arrivals (~1 Hz).
    - One visible canvas sized to the wrapper element. Each `drawImage(backing, dx, dy, dw, dh)` composites the backing onto the visible canvas with a transform that translates by `info.origin.position.{x,y}` * pxPerMeter and scales by `info.resolution * pxPerMeter` (px/cell = m/cell × px/m). Plus flip Y (ROS +y north vs canvas +y south).
    - Robot at canvas center, same math as ScanCanvas standalone projector (canvas center = robot origin; `pxPerMeter = (fit / effMax) * view.zoom`, `fit = min(w,h) * STANDALONE_FIT_FACTOR`, `effMax = EFFECTIVE_RANGE_M = 15 m`). Use constants from ScanCanvas — import them if they're exported, or re-declare as a small local constants block with a comment pointing at scan-canvas.tsx as the source of truth.
    - React to `useViewStore` (zoom/pan/tick) for the visible drawImage transform — re-composite on every tick. The expensive `putImageData` stays keyed on `useMapStore(s => s.latest)` identity only.
    - Defensive bounds check (STRIDE T-04-02-03 below): if `data.length !== info.width * info.height`, skip render and `console.warn` once per identity. Also skip if `info.width <= 0` or `info.height <= 0`.

    Colormap (per RESEARCH.md lines ~238-246, Nav2/rviz convention):
    - `v < 0` (unknown): RGB(96, 96, 96), alpha 180 — dim grey (RESEARCH uses 80; bumping to 96 keeps it visible on the black page bg but still clearly "unknown")
    - `v === 0` (free, high confidence): RGB(240, 240, 240)
    - `v >= 65` (occupied): RGB(20, 20, 20) — near-black
    - middle: linear interp from 200 down to 20 as v goes 1..100 (roughly `200 - v * 1.8`)

    z-ordering: MapBitmap's canvas MUST render under ScanCanvas's canvas. The `/lidar` page wrapper places `<MapBitmap />` BEFORE `<ScanCanvas />` in DOM order and/or sets lower z-index. Canvas `position: absolute; inset: 0` for both.

    Skeleton:
    ```tsx
    "use client";
    import { useEffect, useRef } from "react";
    import { useMapStore } from "@/lib/store/map-store";
    import { useViewStore } from "@/lib/store/view-store";

    // Must match ScanCanvas standalone projector — see scan-canvas.tsx for source of truth.
    const STANDALONE_FIT_FACTOR = 0.45;
    const EFFECTIVE_RANGE_M = 15.0;

    export function MapBitmap({ className }: { className?: string }) {
      const wrapperRef = useRef<HTMLDivElement | null>(null);
      const visibleRef = useRef<HTMLCanvasElement | null>(null);
      const backingRef = useRef<HTMLCanvasElement | null>(null);
      const latest = useMapStore((s) => s.latest);
      const zoom = useViewStore((s) => s.zoom);
      const panX = useViewStore((s) => s.panX);
      const panY = useViewStore((s) => s.panY);

      // Repaint backing canvas on new map identity (putImageData is the expensive step)
      useEffect(() => {
        if (!latest) return;
        const { info, data } = latest;
        if (info.width <= 0 || info.height <= 0) return;
        if (data.length !== info.width * info.height) {
          console.warn("[map-bitmap] data length mismatch", data.length, info.width * info.height);
          return;
        }
        let backing = backingRef.current;
        if (!backing || backing.width !== info.width || backing.height !== info.height) {
          backing = document.createElement("canvas");
          backing.width = info.width;
          backing.height = info.height;
          backingRef.current = backing;
        }
        const bctx = backing.getContext("2d");
        if (!bctx) return;
        const img = bctx.createImageData(info.width, info.height);
        for (let i = 0; i < data.length; i++) {
          const v = data[i];
          let r = 96, g = 96, b = 96, a = 255;
          if (v < 0)       { r = g = b = 96;  a = 180; }
          else if (v === 0){ r = g = b = 240; }
          else if (v >= 65){ r = g = b = 20;  }
          else             { const k = Math.round(200 - v * 1.8); r = g = b = k; }
          const o = i * 4;
          img.data[o] = r; img.data[o+1] = g; img.data[o+2] = b; img.data[o+3] = a;
        }
        bctx.putImageData(img, 0, 0);
      }, [latest]);

      // Composite backing -> visible on every view tick (or new map)
      useEffect(() => {
        const canvas = visibleRef.current;
        const backing = backingRef.current;
        if (!canvas) return;
        const ctx = canvas.getContext("2d");
        if (!ctx) return;
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        if (!backing || !latest) return;
        const { info } = latest;
        const w = canvas.width;
        const h = canvas.height;
        const fit = Math.min(w, h) * STANDALONE_FIT_FACTOR;
        const effMax = EFFECTIVE_RANGE_M;
        const pxPerMeter = (fit / effMax) * zoom;
        const cx = w / 2 + panX;
        const cy = h / 2 + panY;
        // info.origin.position is the map-frame pose of cell (0,0) in meters.
        // Robot is at map-frame origin (approximately — slam_toolbox publishes map->odom
        // such that the robot sits near map origin post-reset). We render the bitmap
        // anchored at info.origin.position, with each cell = info.resolution meters.
        const cellPx = info.resolution * pxPerMeter;
        const dw = info.width * cellPx;
        const dh = info.height * cellPx;
        // Flip Y: ROS +y north, canvas +y south.
        const dx = cx + info.origin.position.x * pxPerMeter;
        const dy = cy - (info.origin.position.y * pxPerMeter + dh);
        ctx.imageSmoothingEnabled = false;
        ctx.drawImage(backing, dx, dy, dw, dh);
      }, [latest, zoom, panX, panY]);

      // Resize handling — same pattern as ScanCanvas
      useEffect(() => {
        const wrapper = wrapperRef.current;
        const canvas = visibleRef.current;
        if (!wrapper || !canvas) return;
        const resize = () => { canvas.width = wrapper.clientWidth; canvas.height = wrapper.clientHeight; };
        resize();
        const ro = new ResizeObserver(resize);
        ro.observe(wrapper);
        return () => ro.disconnect();
      }, []);

      return (
        <div ref={wrapperRef} className={className} style={{ position: "absolute", inset: 0, pointerEvents: "none" }}>
          <canvas ref={visibleRef} style={{ position: "absolute", inset: 0, pointerEvents: "none" }} />
        </div>
      );
    }
    ```
