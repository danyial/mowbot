    **Flag G — v0 anchor limitation, called out in a code comment:**
    v0 assumes the robot sits near the map-frame origin (true immediately after `/slam_toolbox/reset`). `info.origin.position` is the pose of cell (0,0) in the map frame. We render the bitmap anchored at `info.origin.position`, with each cell = `info.resolution` meters. As the mower moves, scan (in base_link frame) and bitmap (in map frame) will diverge — tracking robot-in-map via `/tf` or `/odometry/filtered` is a v1 follow-up. Under SC#4 (stationary) and P1 (freshly-reset map) this limitation is invisible.

    Colormap (Nav2/rviz convention):
    - `v < 0` (unknown): RGB(96, 96, 96), alpha 180
    - `v === 0` (free): RGB(240, 240, 240)
    - `v >= 65` (occupied): RGB(20, 20, 20)
    - middle: linear interp ~`200 - v * 1.8`

    Skeleton:
    ```tsx
    "use client";
    import { useEffect, useRef } from "react";
    import { useMapStore } from "@/lib/store/map-store";
    import type { ScanUnderlayTransform } from "@/components/lidar/scan-canvas";

    export interface MapBitmapProps {
      transform: ScanUnderlayTransform;
    }

    /**
     * v0 anchor limitation: this component renders the map-frame OccupancyGrid
     * anchored at `info.origin.position`, with the robot assumed to be at the
     * map-frame origin (true immediately post-reset). Once the mower moves, the
     * scan (base_link frame) and the bitmap (map frame) will drift apart —
     * tracking robot-in-map via /tf is a v1 follow-up.
     */
    export function MapBitmap({ transform }: MapBitmapProps) {
      const visibleRef = useRef<HTMLCanvasElement | null>(null);
      const backingRef = useRef<HTMLCanvasElement | null>(null);
      const latest = useMapStore((s) => s.latest);

      // Repaint backing on new map identity (expensive).
      useEffect(() => {
        if (!latest) { backingRef.current = null; return; }
        const { info, data } = latest;
        if (info.width <= 0 || info.height <= 0) return;
        if (data.length !== info.width * info.height) {
          console.warn("[map-bitmap] data length mismatch", data.length, info.width * info.height);
          return;
        }
        let backing = backingRef.current;
        if (!backing || backing.width !== info.width || backing.height !== info.height) {
          backing = document.createElement("canvas");
          backing.width = info.width;
          backing.height = info.height;
          backingRef.current = backing;
        }
        const bctx = backing.getContext("2d");
        if (!bctx) return;
        const img = bctx.createImageData(info.width, info.height);
        for (let i = 0; i < data.length; i++) {
          const v = data[i];
          let r = 96, g = 96, b = 96, a = 255;
          if (v < 0)        { r = g = b = 96;  a = 180; }
          else if (v === 0) { r = g = b = 240; }
          else if (v >= 65) { r = g = b = 20;  }
          else              { const k = Math.round(200 - v * 1.8); r = g = b = k; }
          const o = i * 4;
          img.data[o] = r; img.data[o+1] = g; img.data[o+2] = b; img.data[o+3] = a;
        }
        bctx.putImageData(img, 0, 0);
      }, [latest]);

      // Composite backing -> visible on every transform prop change OR new map.
      useEffect(() => {
        const canvas = visibleRef.current;
        if (!canvas) return;
        // Keep visible canvas sized to the transform's canvasWidth/Height (provided
        // by ScanCanvas which owns ResizeObserver).
        if (canvas.width !== transform.canvasWidth) canvas.width = transform.canvasWidth;
        if (canvas.height !== transform.canvasHeight) canvas.height = transform.canvasHeight;
        const ctx = canvas.getContext("2d");
        if (!ctx) return;
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        const backing = backingRef.current;
        if (!backing || !latest) return;
        const { info } = latest;
        const { pxPerMeter, panX, panY, canvasWidth, canvasHeight } = transform;
        const cx = canvasWidth / 2 + panX;
        const cy = canvasHeight / 2 + panY;
    ```
