    // If the project name differs at runtime (e.g. parent-dir-derived), allow override via env:
    const LABEL_FILTER = process.env.MOWER_COMPOSE_LABEL || PROJECT_LABEL;

               follow: true, stdout: true, stderr: true, tail, timestamps: true,

             stderrPT.on("data", emit("stderr"));

               container.modem.demuxStream(raw, stdoutPT, stderrPT);
             } else {
               raw.pipe(stdoutPT);
             }

             raw.on("error", (e) => {
               console.error("[logs-stream] upstream error:", e.message);
               if (clientWs.readyState === clientWs.OPEN) clientWs.close(1011);
             });
             raw.on("end", () => {
               if (clientWs.readyState === clientWs.OPEN) clientWs.close(1000);
             });
           } catch (err) {

             if (clientWs.readyState === clientWs.OPEN) clientWs.close(1011);
           }

             console.log("[logs-stream] Client disconnected from", id);
             try { raw && raw.destroy(); } catch {}
             try { stdoutPT && stdoutPT.destroy(); } catch {}
             try { stderrPT && stderrPT.destroy(); } catch {}
           });
         });
