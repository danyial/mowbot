```typescript
/**
 * Extract yaw directly in radians from a quaternion. Used by map-frame
 * cursor heading and the computeMapFramePose helper — both need radians
 * for trig without a deg↔rad round-trip. Formula is identical to
 * quaternionToEuler's yaw extraction (atan2 of z-component).
 *
 * Convention: CCW from +x axis (math convention, matches ROS REP-103).
 */
export function quaternionToYawRad(q: Quaternion): number {
  return Math.atan2(
    2 * (q.w * q.z + q.x * q.y),
    1 - 2 * (q.y * q.y + q.z * q.z)
  );
}
```

```javascript
/**
 * computeMapFramePose — SE(2) composition of slam anchor + odom delta.
 *
 * Given a slam-pose anchor captured at time T (with the matching odom
 * snapshot at T) and the current odom reading, produce the live map-frame
 * base_link pose. The anchor odom is used to compute odom deltas; the
 * slam anchor's yaw vs anchorOdom.yaw tells us how to rotate the odom
 * delta into the map frame.
 *
 * All angles are radians.
 *
 * @param {{ x:number, y:number, yaw:number, anchorOdom: {x:number,y:number,yaw:number}|null } | null} slam
 * @param {{ posX:number, posY:number, yawRad:number }} odom
 * @returns {{ x:number, y:number, yaw:number } | null} map-frame pose, or null when no slam anchor is available
 *
 * See 08-RESEARCH §Composite Pose Math lines 556-623 for derivation.
 */
export function computeMapFramePose(slam, odom) {
  if (!slam || !slam.anchorOdom) return null;
  const dx = odom.posX - slam.anchorOdom.x;
  const dy = odom.posY - slam.anchorOdom.y;
  const dyaw = odom.yawRad - slam.anchorOdom.yaw;
  const dtheta = slam.yaw - slam.anchorOdom.yaw;
  const cos = Math.cos(dtheta);
  const sin = Math.sin(dtheta);
  const dxMap = cos * dx - sin * dy;
  const dyMap = sin * dx + cos * dy;
  return {
    x: slam.x + dxMap,
    y: slam.y + dyMap,
    yaw: slam.yaw + dyaw,
  };
}
```

```typescript
function isBrowser() {

function isQuotaExceededError(err: unknown): boolean {
  // Cross-browser handling (Chrome, Firefox legacy, Safari). See
  // https://mmazzarolo.com/blog/2022-06-25-local-storage-status
  if (err instanceof DOMException) {
    if (
      err.code === 22 ||
      err.code === 1014 ||
      err.name === "QuotaExceededError" ||
      err.name === "NS_ERROR_DOM_QUOTA_REACHED"
    ) {
      return true;
    }
  }
  // Node:test shim path — plain object with name/code set
  const e = err as { name?: string; code?: number };
  if (e && (e.name === "QuotaExceededError" || e.code === 22)) return true;
  return false;
}

function safeSetItem(key: string, value: string): "ok" | "quota" | "error" {
  if (!isBrowser()) return "error";
  try {
    localStorage.setItem(key, value);
    return "ok";
  } catch (err) {
    if (isQuotaExceededError(err)) return "quota";
    console.error("[map-store] localStorage.setItem error:", err);
    return "error";
  }
}

export const useMapStore = create<MapState>((set, get) => ({
  latest: null,
  lastMessageAt: null,
  isStale: true,
  epoch: null,
  persistenceDisabled: false,

  updateMap: (m) => {
    set({ latest: m, lastMessageAt: Date.now(), isStale: false });
    const { epoch, persistenceDisabled } = get();
    if (!isBrowser() || epoch == null || persistenceDisabled) return;
    // Int8Array round-trip (Pitfall 5): JSON cannot represent typed arrays.
    // Convert to plain array on write; rehydrate converts back.
    const serializable = { ...m, data: Array.from(m.data as unknown as Int8Array) };
    const result = safeSetItem(`mowerbot.map.epoch.${epoch}`, JSON.stringify(serializable));
    if (result === "quota") {
      set({ persistenceDisabled: true });
      console.warn("[map-store] localStorage quota exceeded; persistence disabled for session");
    }
  },

  setStale: (s) => set({ isStale: s }),
  clear: () => set({ latest: null, lastMessageAt: null, isStale: true }),

  setEpoch: (n) => {
    set({ epoch: n });
    if (!isBrowser()) return;
    // Drop keys with suffix < n (int compare, not string — else "10" < "9")
    const prefix = "mowerbot.map.epoch.";
    const toRemove: string[] = [];
    for (let i = 0; i < localStorage.length; i++) {
      const k = localStorage.key(i);
      if (!k || !k.startsWith(prefix)) continue;
      const suffix = parseInt(k.slice(prefix.length), 10);
      // NaN or < n → remove. This cleans up pre-Phase-8 debug keys too.
      if (!Number.isFinite(suffix) || suffix < n) toRemove.push(k);
    }
    for (const k of toRemove) {
      try { localStorage.removeItem(k); } catch { /* noop */ }
    }
  },

  rehydrate: async () => {
    if (!isBrowser()) return;
    try {
      const res = await fetch("/api/map/epoch", { cache: "no-store" });
      if (!res.ok) throw new Error(`GET /api/map/epoch ${res.status}`);
      const payload = await res.json();
      if (typeof payload.epoch !== "number") throw new Error("malformed epoch response");
      const epoch = payload.epoch;
      const raw = localStorage.getItem(`mowerbot.map.epoch.${epoch}`);
      if (raw) {
        try {
          const parsed = JSON.parse(raw);
          const hydrated: OccupancyGrid = {
            ...parsed,
            data: Int8Array.from(parsed.data),
          };
          set({ latest: hydrated, lastMessageAt: null, isStale: true, epoch });
        } catch (err) {
          console.error("[map-store] rehydrate parse failed:", err);
          set({ epoch });
        }
      } else {
        set({ epoch });
      }
      // Cleanup stale keys (and pre-Phase-8 debug keys) using the same
      // setEpoch path — keeps the cleanup logic in one place.
      get().setEpoch(epoch);
    } catch (err) {
      console.error("[map-store] rehydrate failed:", err);
    }
  },
}));
```
