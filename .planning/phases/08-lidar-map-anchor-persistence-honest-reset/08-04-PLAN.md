```typescript
onClick={async () => {
  // D-16 step 1: optimistic clear so the bitmap wipes immediately
  useMapStore.getState().clear();
  try {
    const res = await fetch("/api/map/reset", { method: "POST" });
    const body = await res.json();
    if (body.ok) {
      // Happy path: update epoch, show toast (auto-dismiss 3s)
      useMapStore.getState().setEpoch(body.epoch);
      setResetFeedback({ kind: "toast", text: "Karte zurückgesetzt" });
    } else if (body.stage === "service") {
      // D-16 step 5: red banner + Retry. Revert optimistic clear by
      // re-running rehydrate from localStorage (D-17 constraint: service
      // stage is the ONLY failure that reverts).
      useMapStore.getState().rehydrate();
      setResetFeedback({
        kind: "banner-red",
        text: body.error || "Reset fehlgeschlagen",
      });
    } else if (body.stage === "mapTimeout") {
      // D-16 step 6: epoch IS bumped; leave map cleared; yellow banner.
      useMapStore.getState().setEpoch(body.epoch);
      setResetFeedback({
        kind: "banner-yellow",
        text: body.error || "Reset durchgeführt, aber slam_toolbox publisht keine neue Karte.",
      });
    } else {
      setResetFeedback({ kind: "banner-red", text: "Unerwartete Antwort vom Server" });
    }
  } catch (e) {
    console.error("[lidar] /api/map/reset failed:", e);
    useMapStore.getState().rehydrate();
    setResetFeedback({
      kind: "banner-red",
      text: String(e instanceof Error ? e.message : e),
    });
  }
}}
```
